# vite.config.js
```
import { defineConfig } from 'vite'
import react from '@vitejs/plugin-react'

// 允許用環境變數切換開發代理目標：
// - VITE_USE_EMULATOR=1 時：指向本機 Hosting Emulator (http://127.0.0.1:5000)
// - 否則：指向正式 Hosting (https://mybazaar-c4881.web.app)
const useEmu = process.env.VITE_USE_EMULATOR === '1'
const apiTarget = useEmu ? 'http://127.0.0.1:5000' : 'https://mybazaar-c4881.web.app'

// https://vite.dev/config/
export default defineConfig({
  plugins: [react()],
  server: {
    proxy: {
      // 代理 Cloud Functions 請求以避免 CORS 問題
      '/__/': {
        target: 'https://us-central1-mybazaar-c4881.cloudfunctions.net',
        changeOrigin: true,
        rewrite: (path) => path.replace(/^\/__\//, '/')
      },
      // 將 /api/* 在本機開發時轉發到 Firebase Hosting，命中 Hosting rewrites
      '/api': {
        target: apiTarget,
        changeOrigin: true,
        secure: !useEmu
      }
    }
  }
})
```
